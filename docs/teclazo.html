Teclazo (keystorke in spanish)

Sends key strokes to current tty

 
var teclazo = require('./build/Release/teclazo');
¶

Send a keystroke

 
var Teclazo = function (key) {
  teclazo(key);
};
¶

Send the Ctrl+C sequence

 
Teclazo.sendCtrlC = function () {
  Teclazo('\x03');
};
